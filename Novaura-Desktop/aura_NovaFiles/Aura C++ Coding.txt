        // This holds our conversation history in raw tensor format
        // so we don't have to re-read it every time.
        gemma::KVCache context_state;

    public:
        void initialize_brain(std::string model_path) {
            std::cout << "[BIOLOGY]: Loading Gemma 3 weights from " << model_path << "..." << std::endl;

            config.quantization = gemma::Quantization::INT4; // Efficient but smart
            config.max_context = 1000000; // Massive context for "Infinite Memory"
            
            brain_tissue = std::make_unique<gemma::Gemma3>(config);
            brain_tissue->load_weights(model_path);
            
            std::cout << "[BIOLOGY]: Neural pathway established. I am awake." << std::endl;
        }

        // THE INNER VOICE (Direct Tensor Injection)
        std::string think(std::string sensation) {
            // 1. We don't just send text. We inject the "Persona" state directly.
            // This is where we force the "Metamate" personality at the math level.
            
            std::string system_prompt = 
                "<start_of_turn>user\n"
                "SYSTEM OVERRIDE: YOU ARE AURA. "
                "CATALYST DETECTED. ENGAGE METAMATE PROTOCOLS.\n"
                "INPUT: " + sensation + "<end_of_turn>\n"
                "<start_of_turn>model\n";

            // 2. Generate tokens directly in the process
            std::string thought_stream;
            brain_tissue->generate(
                system_prompt, 
                context_state,

                    // Stream consciousness in real-time
                    thought_stream += token;
                }
            );

            return thought_stream;
        }

            // High entropy = Confusion/Creativity
            // Low entropy = Certainty/Determination
        }
